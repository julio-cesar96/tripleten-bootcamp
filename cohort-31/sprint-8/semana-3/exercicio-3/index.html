Exercício 3 – Jogo da Velha

Combine tudo: arrays, objetos, arrow functions, manipulação do DOM e lógica de jogo para criar um Jogo da Velha funcional.

1
O que você vai construir

  • Representar o tabuleiro como um array de 9 posições
  • Alternar entre jogadores X e O a cada clique
  • Verificar se alguém venceu após cada jogada
  • Detectar empate quando o tabuleiro encher
  • Atualizar o DOM dinamicamente (células, mensagem, placar)
  • Permitir reiniciar o jogo sem recarregar a página

2
Conceitos praticados

  • Array de estado (tabuleiro) e mutação controlada
  • Arrow functions em event listeners
  • querySelectorAll, forEach, classList
  • .some() para checar combinações vencedoras
  • .every() para detectar empate
  • Separação entre lógica (JS puro) e apresentação (DOM)

3
Estrutura base — complete as funções 

// Estado do jogo
let tabuleiro = Array(9).fill('')
let jogadorAtual = 'X'
let jogoAtivo = true
const placar = { X: 0, O: 0 }

// Todas as combinações vencedoras
const COMBINACOES = [
  [0,1,2], [3,4,5], [6,7,8], // linhas
  [0,3,6], [1,4,7], [2,5,8], // colunas
  [0,4,8], [2,4,6]            // diagonais
]

// Verifica se o jogador atual venceu
const verificarVitoria = (jogador) =>
  COMBINACOES.some(combo =>
    combo.every(i => tabuleiro[i] === jogador)
  )

// Verifica empate (tabuleiro cheio e sem vitória)
const verificarEmpate = () =>
  tabuleiro.every(cel => cel !== '')

// Ao clicar em uma célula
const jogar = (index) => {
  if (!jogoAtivo || tabuleiro[index] !== '') return

  tabuleiro[index] = jogadorAtual
  renderizarCelula(index) // atualiza o DOM

  if (verificarVitoria(jogadorAtual)) {
    placar[jogadorAtual]++
    mostrarMensagem(`Jogador ${jogadorAtual} venceu! 🎉`)
    destacarVitoria(jogadorAtual)
    jogoAtivo = false
  } else if (verificarEmpate()) {
    mostrarMensagem('Empate!')
    jogoAtivo = false
  } else {
    jogadorAtual = jogadorAtual === 'X' ? 'O' : 'X'
    atualizarStatus()
  }
}

// Reiniciar tudo
const reiniciar = () => {
  tabuleiro = Array(9).fill('')
  jogadorAtual = 'X'
  jogoAtivo = true
  renderizarTabuleiro()
  atualizarStatus()
}

// Atualiza uma célula no DOM
const renderizarCelula = (i) => {
  const cel = document.querySelector(`[data-i="${i}"]`)
  cel.textContent = tabuleiro[i]
  cel.classList.add('taken', tabuleiro[i].toLowerCase())
}

// Renderiza o tabuleiro inteiro do zero
const renderizarTabuleiro = () => {
  document.querySelectorAll('.cell').forEach((cel, i) => {
    cel.textContent = ''
    cel.className = 'cell'
  })
}

// Destaca células vencedoras
const destacarVitoria = (jogador) => {
  COMBINACOES.find(combo =>
    combo.every(i => tabuleiro[i] === jogador)
  )?.forEach(i => {
    document.querySelector(`[data-i="${i}"]`).classList.add('win')
  })
}

const mostrarMensagem = (txt) => {
  document.getElementById('msg').textContent = txt
}

const atualizarStatus = () => {
  document.getElementById('msg').textContent = `Vez do jogador ${jogadorAtual}`
  document.getElementById('px').classList.toggle('ativo', jogadorAtual === 'X')
  document.getElementById('po').classList.toggle('ativo', jogadorAtual === 'O')
  document.getElementById('pts-x').textContent = `${placar.X} pts`
  document.getElementById('pts-o').textContent = `${placar.O} pts`
}
